#define STACK_SIZE 1000
#define MAX_FRAMES 100

// Stack memory
int stack[STACK_SIZE];
int SP = STACK_SIZE;  // Stack pointer (grows down)
int FP = STACK_SIZE;  // Frame pointer

// Frame information for visualization
typedef struct {
    char funcName[50];
    int basePtr;
    int numLocals;
    int numParams;
} FrameInfo;

FrameInfo frames[MAX_FRAMES];
int frameCount = 0;

// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
// Stack Operations
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

void push(int value) {
    SP--;
    stack[SP] = value;
    printf("    PUSH %d (SP = %d)\n", value, SP);
}

int pop() {
    int value = stack[SP];
    SP++;
    printf("    POP %d (SP = %d)\n", value, SP);
    return value;
}

void allocateLocals(int n) {
    SP -= n;
    printf("    Allocate %d locals (SP = %d)\n", n, SP);
}

void deallocateLocals(int n) {
    SP += n;
    printf("    Deallocate %d locals (SP = %d)\n", n, SP);
}

// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
// Function Call Simulation
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

void callFunction(char* funcName, int* args, int numArgs, int numLocals) {
    printf("\nâ”€â”€ Call %s â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€\n", funcName);
    
    // Caller pushes arguments (right to left)
    printf("  Caller:\n");
    for(int i = numArgs - 1; i >= 0; i--) {
        push(args[i]);
    }
    
    // Push return address (simulated as -1)
    push(-1);  // Return address placeholder
    
    // Callee prologue
    printf("  Callee prologue:\n");
    push(FP);  // Save old frame pointer
    FP = SP;   // Set new frame pointer
    printf("    Set FP = %d\n", FP);
    
    // Allocate space for locals
    allocateLocals(numLocals);
    
    // Save frame info
    strcpy(frames[frameCount].funcName, funcName);
    frames[frameCount].basePtr = FP;
    frames[frameCount].numLocals = numLocals;
    frames[frameCount].numParams = numArgs;
    frameCount++;
    
    printStack();
}

int returnFromFunction() {
    if(frameCount == 0) {
        printf("Error: No function to return from\n");
        return 0;
    }
    
    frameCount--;
    FrameInfo* frame = &frames[frameCount];
    
    printf("\nâ”€â”€ Return from %s â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€\n", frame->funcName);
    printf("  Callee epilogue:\n");
    
    // Deallocate locals
    deallocateLocals(frame->numLocals);
    
    // Restore frame pointer
    int oldFP = pop();
    FP = oldFP;
    printf("    Restore FP = %d\n", FP);
    
    // Pop return address
    pop();  // Return address
    
    // Caller cleans up arguments
    printf("  Caller cleanup:\n");
    for(int i = 0; i < frame->numParams; i++) {
        pop();
    }
    
    printStack();
    return 0;
}

void setLocal(int offset, int value) {
    stack[FP - 1 - offset] = value;
    printf("    Set local[%d] = %d\n", offset, value);
}

int getLocal(int offset) {
    return stack[FP - 1 - offset];
}

int getParam(int offset) {
    // Parameters are above saved FP and return address
    return stack[FP + 2 + offset];

    int currentFrame = frameCount - 1;
    int frameBoundary = currentFrame >= 0 ? frames[currentFrame].basePtr : STACK_SIZE;
    
    for(int i = SP; i < STACK_SIZE && i < SP + 20; i++) {
        char marker[20] = "";
        
        if(i == SP) strcat(marker, " <-SP");
        if(i == FP) strcat(marker, " <-FP");
        
        // Identify content
        char content[30] = "";
        if(currentFrame >= 0) {
            int localStart = frames[currentFrame].basePtr - 1;
            int localEnd = localStart - frames[currentFrame].numLocals + 1;
            
            if(i >= localEnd && i <= localStart) {
                sprintf(content, "local[%d]", localStart - i);
            } else if(i == frames[currentFrame].basePtr) {
                sprintf(content, "saved FP");
            } else if(i == frames[currentFrame].basePtr + 1) {
                sprintf(content, "ret addr");
            } else if(i > frames[currentFrame].basePtr + 1) {
                sprintf(content, "param[%d]", i - frames[currentFrame].basePtr - 2);
            }
        }
        
        printf("  â”‚  %4d   â”‚  %6d %-14s â”‚%s\n", 
               i, stack[i], content, marker);
    }
