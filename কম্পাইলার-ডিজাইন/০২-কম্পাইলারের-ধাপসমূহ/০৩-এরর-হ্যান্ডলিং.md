    LEXICAL_ERROR,
    SYNTAX_ERROR,
    SEMANTIC_ERROR,
    WARNING
} ErrorType;

// Error structure
typedef struct {
    ErrorType type;
    int line;
    int column;
    char message[256];
    char suggestion[128];
} CompilerError;

// Error list
CompilerError errors[100];
int errorCount = 0;

// Add error
void addError(ErrorType type, int line, int col, 
              char* msg, char* suggestion) {
    if(errorCount < 100) {
        errors[errorCount].type = type;
        errors[errorCount].line = line;
        errors[errorCount].column = col;
        strcpy(errors[errorCount].message, msg);
        if(suggestion)
            strcpy(errors[errorCount].suggestion, suggestion);
        else
            errors[errorCount].suggestion[0] = '\0';
        errorCount++;
    }
}

// Get error type string
char* getErrorTypeString(ErrorType type) {
    switch(type) {
        case LEXICAL_ERROR: return "Lexical Error";
        case SYNTAX_ERROR: return "Syntax Error";
        case SEMANTIC_ERROR: return "Semantic Error";
        case WARNING: return "Warning";
        default: return "Unknown Error";
    }
}
