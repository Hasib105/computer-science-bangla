#include <stdlib.h>
#include <string.h>
#include <ctype.h>

// Simple regex matcher for identifiers
int matchIdentifier(char* str) {
    if(str == NULL || *str == '\0') return 0;
    
    // First character: letter or underscore
    if(!isalpha(*str) && *str != '_') return 0;
    str++;
    
    // Remaining: letter, digit, or underscore
    while(*str != '\0') {
        if(!isalnum(*str) && *str != '_') return 0;
        str++;
    }
    
    return 1;
}

// Simple regex matcher for integers
int matchInteger(char* str) {
    if(str == NULL || *str == '\0') return 0;
    
    // Optional sign
    if(*str == '+' || *str == '-') str++;
    
    if(*str == '\0') return 0;
    
    // At least one digit
    while(*str != '\0') {
        if(!isdigit(*str)) return 0;
        str++;
    }
    
    return 1;
}

// Simple regex matcher for floats
int matchFloat(char* str) {
    if(str == NULL || *str == '\0') return 0;
    
    // Optional sign
    if(*str == '+' || *str == '-') str++;
    
    int hasDigit = 0;
    int hasDot = 0;
    
    // Integer part
    while(isdigit(*str)) {
        hasDigit = 1;
        str++;
    }
    
    // Decimal point
    if(*str == '.') {
        hasDot = 1;
        str++;
    }
    
    // Fractional part
    while(isdigit(*str)) {
        hasDigit = 1;
        str++;
    }
    
    return hasDigit && hasDot && *str == '\0';
}

int main() {
    // Test Identifier
    printf("Identifier Tests:\n");
    printf("  'count': %s\n", matchIdentifier("count") ? "Match" : "No Match");
    printf("  '_var': %s\n", matchIdentifier("_var") ? "Match" : "No Match");
    printf("  '2name': %s\n", matchIdentifier("2name") ? "Match" : "No Match");
    
    // Test Integer
    printf("\nInteger Tests:\n");
    printf("  '123': %s\n", matchInteger("123") ? "Match" : "No Match");
    printf("  '-42': %s\n", matchInteger("-42") ? "Match" : "No Match");
    printf("  '12.5': %s\n", matchInteger("12.5") ? "Match" : "No Match");
    
    // Test Float
    printf("\nFloat Tests:\n");
    printf("  '3.14': %s\n", matchFloat("3.14") ? "Match" : "No Match");
    printf("  '0.5': %s\n", matchFloat("0.5") ? "Match" : "No Match");
